# 深浅拷贝

# 浅拷贝

是指只复制第一层对象，但是当对象的属性是引用类型时，实质复制的是其引用，当引用指向的值改变时也会跟着变化。

1.Object.assign()

拷贝所有的属性值到新的对象中，如果属性值是对象的话，拷贝的是地址

属性被后续参数中具有相同属性的其他对象覆盖。

const target = { a: 1, b: 2 };
const source1 = { b: 4, c: 5 };
const source2 = { b: 6 };
const returnedTarget = Object.assign(target, source,source2);
console.log(target);// {a: 1, b: 6, c: 5}
console.log(returnedTarget);//  Object {a: 1, b: 6, c: 5}
console.log(target===returnedTarget);//true

2.展开运算符 ...

let a = {
  age: 1
}
let b = { ...a }
a.age = 2
console.log(b.age) // 1

# 深拷贝

深拷贝后的对象与原来的对象是完全隔离的，互不影响

1.过 JSON.parse(JSON.stringify(object)) 。

let a = {
  age: 1,
  jobs: {
    first: 'FE'
  }
}
let b = JSON.parse(JSON.stringify(a))
a.jobs.first = 'native'
console.log(b.jobs.first) // FE

局限性的：

  • 会忽略 undefined symbol函数
  • 不能解决循环引用的对象
let a = {
  age: undefined,
  sex: Symbol('male'),
  jobs: function() {},
  name: 'yck'
}
let b = JSON.parse(JSON.stringify(a))
console.log(b) // {name: "yck"}

let obj = {
  a: 1,
  b: {
    c: 2,
    d: 3,
  },
}
obj.c = obj.b
obj.e = obj.a
obj.b.c = obj.c
obj.b.d = obj.b
obj.b.e = obj.b.c
let newObj = JSON.parse(JSON.stringify(obj))
console.log(newObj)
//TypeError: Converting circular structure to JSON

如果你所需拷贝的对象含有内置类型并且不包含函数，可以使用 MessageChannel

function structuralClone(obj) {
  return new Promise(resolve => {
    const { port1, port2 } = new MessageChannel()
    port2.onmessage = ev => resolve(ev.data)
    port1.postMessage(obj)
  })
}

var obj = {
  a: 1,
  b: {
    c: 2
  }
}

obj.b.d = obj.b

// 注意该方法是异步的
// 可以处理 undefined 和循环引用对象
const test = async () => {
  const clone = await structuralClone(obj)
  console.log(clone)
}
test()

# 深拷贝简易版

//方法
function deepClone(obj) {
    if(typeof obj !== 'object' || obj === null) {
        return obj;
    }
    //如果obj是数组，那么 obj.constructor 是 [Function: Array]
    //如果obj是对象，那么 obj.constructor 是 [Function: Object]
    let t = new obj.constructor();
    for(let key in obj) {
        t[key] = deepClone(obj[key]);
    }
    return t;
}
//验证
let obj = {
  a: [1, 2, 3],
  b: {
    c: 2,
    d: 3
  },
  e:null
}
let newObj = deepClone(obj)
newObj.a[2] = 1
newObj.b.c = 1
newObj.e = 1
console.log(obj)
console.log(newObj)

# for-in和for-of的区别

for-in遍历的是key，for-of遍历的是value，都可以使用break、continue和return。

# for-in 不推荐用来遍历数组

1.index索引为字符串型，进行几何运算注意。

2.遍历顺序有可能不是按照实际数组的内部顺序。

3.使用for-in会遍历所有的可枚举属性，包括原型。通常需要配合hasOwnProperty()方法判断某个属性是否该对象的实例属性，来将原型对象从循环中剔除。

var triangle = {a: 1, b: 2, c: 3};
triangle.__proto__.name = 'le'
for (var prop in triangle) {
  if (triangle.hasOwnProperty(prop)) {
   console.log(prop)
  } 
}
# for-of可以简单、正确地遍历数组。（普通对象没有iterable无法使用，可以用于遍历Map和Set）
var myArray = [1, 2, 4, 5, 6, 7];
myArray.name = "数组";
for (var value of myArray) {
    console.log(value);
}

var set = new Set();
set.add("a").add("b");
for (let s of set) {
    console.log(s);
}

var map = new Map();
map.set("a", 1).set("b", 2);
for(let [k, v] of map) {
    console.log(k, v);
}

# Proxy

let p = new Proxy(target, handler)

target 代表需要添加代理的对象，handler是一个对象，定义了各种操作触发的函数，比如set或者get` 函数。

// 用Proxy包装对象，实现监听
  let handler = {
    get(target, property, receiver) {
      console.log(`'${property}' = ${target[property]}`)
      return target[property]
    },
    set(target, property, value, receiver) {
      console.log(`监听到属性${property}改变为${value}`)
      target[property] = value
      return true
    }
  }

let obj = { a: 1 }
let p = new Proxy(obj, handler)
p.a = 2 // 监听到属性a改变
p.a // 'a' = 2
//原对象依然可以操作，但是不会触发监听函数
obj.a = 10
p.a //10

Vue3.0 使用 Proxy 替换原本的 API 原因在于 Proxy 无需一层层递归为每个属性添加代理性能上更好，并且原本的实现有一些数据更新不能监听到，但是 Proxy 可以完美监听到任何方式的数据改变，唯一缺陷是浏览器的兼容性不好了。

# 手写apply与call和bind

 Function.prototype.myapply = function (context, arg) {
        var context = context || window
        context.fn = this
        var result = context.fn(...arg)
        delete context.fn
        return result
    }

    Function.prototype.mycall = function (context, ...arg) {
        var context = context || window
        context.fn = this
        var result = context.fn(...arg)
        delete context.fn
        return result
    }

    Function.prototype.mybind = function (context, ...arg) {
        var context = context || window
        var that = this
        return function F() {
            let newArg = [...arguments]
            // 判断是否被当做构造函数使用
            if (this instanceof F) {
                return that.apply(this, arg.concat(newArg))
            }
            return that.apply(context, arg.concat(newArg))
        }
    }

    var a = 1;
    var obj = {
        a: 2
    }
    function test(a, b) {
        console.log(this.a + a + b)
    }
    function test2(a, b) {
        this.a = a
        this.b = b
    }
    test(3, 4)
    test.myapply(null, [3, 4])
    test.myapply(obj, [3, 4])
    test.mycall(null, 3, 4)
    test.mycall(obj, 3, 4)
    var fn = test.mybind(obj, 3)

# 手写promise简易版

    function myPromise(fn) {
        var that = this
        this.state = 'pending'
        this.successCb = []
        this.failCb = []
        this.value = null
        function resolve(value) {
            if (that.state = 'pending') {
                that.state = 'fullfilled'
                that.value = value
                that.successCb.forEach(item => item(value))
            }
        }
        function reject(value) {
            if (that.state = 'pending') {
                that.state = 'rejected'
                that.value = value
                that.failCb.forEach(item => item(value))
            }
        }
        // 接受的函数需要立即执行
        fn(resolve, reject)

    }
    myPromise.prototype.then = function (res, rej) {
        // new promise 传的函数异步时
        if (typeof res === 'function') {
            this.successCb.push(res)
        }
        if (typeof rej === 'function') {
            this.failCb.push(rej)
        }
        // new promise 传的函数同步时，执行then的时候状态已经改变
        if (this.state === 'fullfilled') {
            res(this.value)
        }
        if (this.state === 'rejected') {
            rej(this.value)
        }
    }


    new myPromise(function (resolve, reject) {
        setTimeout(function () {
            resolve('异步')
        }, 1000)
    }).then(function (msg) {
        console.log(msg)
    })

    new myPromise(function (resolve, reject) {
        resolve('同步')
    }).then(function (msg) {
        console.log(msg)
    })

# 节流与防抖

    // 节流
    // 高频事件触发，但在 n 秒内只会执行一次。
    // 在类似监听页面元素滚动事件的时候会用到。因为滚动事件，是一个高频触发的事件。
    function throttle(fn) {
        var flag = true
        return function () {
            if (flag) {
                flag = false
                setTimeout(() => {
                    flag = true
                    fn.apply(this, arguments);
                }, 1000)
            }
        }
    }

    // 防抖
    // 触发高频事件一次后 n 秒后函数执行，如果 n 秒内高频事件再次被触发，则函数不会执行，重新计算时间 n秒 ；
    // 输入框只有等用户输入完毕后，才检查格式是否正确。
    function debounce(fn) {
        var flag = null
        return function () {
            clearTimeout(flag)
            flag = setTimeout(fn.apply(this, arguments), 1000)
        }
    }

# 发布订阅模式

    var obj = {
        list: {},
        on(event, cb) {
            (this.list[event] || (this.list[event] = [])).push(cb)
        },
        emit(event) {
            this.list[event].forEach(fn => {
                fn()
            });
        },
        off(event, cb) {
            // 没传函数，全清空
            if (!cb) {
                this.list[event] = []
                return
            }
            fns = this.list[event]
            // 没找到事件对应的函数，直接返回
            if (!fns) {
                return
            }
            var len = fns.length
            for (let i = 0; i < len; i++) {
                if (fns[i] === cb) {
                    fns.splice(i, 1)
                }
            }
        },
        // 绑定后事件只能触发一次
        once(event, cb) {
            var _this = this
            // 建一个新的函数传给on。执行cb，然后off
            function fn() {
                cb()
                _this.off(event, fn)
            }
            this.on(event, fn)
        }

    }

    obj.on('event1', function () {
        console.log('event1触发')
    })
    obj.on('event2', function () {
        console.log('event2触发')
    })
    obj.once('event2', function () {
        console.log('触发once')
    })
    obj.emit('event1')
    obj.emit('event2')
    obj.emit('event2')
